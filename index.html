const SHEET_ID = '1OzWhkGcYo5vIHvT5SLASQGnZxzJqyW_-hnecSh1WRo8'; function doGet(e) { try { const emp = e.parameter.emp || 'CEM'; const per = e.parameter.per || 'dia'; const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName(emp + '_' + per); let result = {data:[], updatedAt:''}; if (sheet) { const raw = sheet.getRange(1,1).getValue(); if (raw) result = JSON.parse(raw); } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({error: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { const emp = e.parameter.emp; const per = e.parameter.per; const updatedAt = e.parameter.updatedAt; const data = JSON.parse(e.parameter.data); const ss = SpreadsheetApp.openById(SHEET_ID); const sheetName = emp + '_' + per; let sheet = ss.getSheetByName(sheetName); if (!sheet) sheet = ss.insertSheet(sheetName); sheet.getRange(1,1).setValue(JSON.stringify({data, updatedAt})); return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({error: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
